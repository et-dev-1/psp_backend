## Payment Integration

### Stripe Payment Processing

```typescript
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

// Create a payment intent
app.post('/api/payments/create-intent', authenticateToken, async (req, res) => {
  try {
    const userId = req.user?.id;
    const { amount, currency = 'SEK', orderId } = req.body;

    // Validate amount
    if (!amount || amount < 50) {
      return res.status(400).json({ error: 'Invalid amount' });
    }

    // Create payment intent
    const paymentIntent = await stripe.paymentIntents.create({
      amount: Math.round(amount * 100), // Convert to cents
      currency,
      metadata: {
        userId,
        orderId
      }
    });

    res.json({
      clientSecret: paymentIntent.client_secret,
      intentId: paymentIntent.id
    });
  } catch (error) {
    console.error('Stripe error:', error);
    res.status(500).json({ error: 'Payment processing failed' });
  }
});

// Confirm payment and log transaction
app.post('/api/payments/confirm', authenticateToken, async (req, res) => {
  try {
    const userId = req.user?.id;
    const { intentId, orderId } = req.body;

    // Retrieve intent to verify status
    const paymentIntent = await stripe.paymentIntents.retrieve(intentId);

    if (paymentIntent.status !== 'succeeded') {
      return res.status(400).json({ error: 'Payment not completed' });
    }

    // Log transaction
    const connection = await db.getConnection();
    await connection.query(
      `INSERT INTO transactions (user_id, order_id, amount, currency, status, stripe_intent_id)
       VALUES (?, ?, ?, ?, ?, ?)`,
      [userId, orderId, paymentIntent.amount / 100, paymentIntent.currency, 'completed', intentId]
    );

    // Update order status
    await connection.query(
      `UPDATE orders SET payment_status = 'paid' WHERE id = ? AND user_id = ?`,
      [orderId, userId]
    );

    connection.release();

    // Send confirmation email
    await sendOrderConfirmationEmail(userId, orderId);

    // Notify via WebSocket
    io.to(`order:${orderId}`).emit('payment-completed', {
      orderId,
      paymentStatus: 'paid',
      amount: paymentIntent.amount / 100
    });

    res.json({ message: 'Payment confirmed', transactionId: paymentIntent.id });
  } catch (error) {
    console.error('Payment confirmation error:', error);
    res.status(500).json({ error: 'Payment confirmation failed' });
  }
});

// Stripe Webhook (handle payment events)
app.post('/api/webhooks/stripe', async (req, res) => {
  const sig = req.headers['stripe-signature'] as string;
  let event;

  try {
    event = stripe.webhooks.constructEvent(
      req.body,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET!
    );
  } catch (error) {
    console.error('Webhook verification failed:', error);
    return res.status(400).json({ error: 'Invalid signature' });
  }

  // Handle events
  switch (event.type) {
    case 'payment_intent.succeeded':
      const paymentIntent = event.data.object;
      console.log('Payment succeeded:', paymentIntent.id);
      // Update database
      break;

    case 'payment_intent.payment_failed':
      const failedIntent = event.data.object;
      console.log('Payment failed:', failedIntent.id);
      // Update order status to failed
      break;
  }

  res.json({ received: true });
});
```

### Swish Payment (Sweden)

```typescript
// In swish/SwishApi.ts pattern

app.post('/api/payments/swish', authenticateToken, async (req, res) => {
  try {
    const { phoneNumber, amount, orderId } = req.body;
    const userId = req.user?.id;

    // Initialize Swish API (check swish/SwishApi.ts for setup)
    const swishApi = new SwishApi();

    // Create payment request
    const paymentRequest = await swishApi.createPaymentRequest({
      phoneNumber,
      amount,
      message: `Order #${orderId}`,
      callbackUrl: `${process.env.BACKEND_PUBLIC_BASE_URL}/api/webhooks/swish`
    });

    // Log payment attempt
    const connection = await db.getConnection();
    await connection.query(
      `INSERT INTO transactions (user_id, order_id, amount, currency, status, payment_method)
       VALUES (?, ?, ?, ?, ?, ?)`,
      [userId, orderId, amount, 'SEK', 'pending', 'swish']
    );
    connection.release();

    res.json({
      paymentRequestId: paymentRequest.id,
      status: 'pending'
    });
  } catch (error) {
    console.error('Swish error:', error);
    res.status(500).json({ error: 'Swish payment failed' });
  }
});
```

## Email Integration

### Sending Emails with Handlebars Templates

```typescript
// In email/email.ts (or create if not exists)

import nodemailer, { Transporter } from 'nodemailer';
import fs from 'fs';
import path from 'path';
import Handlebars from 'handlebars';

let transporter: Transporter;

// Initialize email transporter
if (process.env.SMTP_TEST_MODE === 'true') {
  // Test mode: log emails instead of sending
  transporter = {
    sendMail: async (options) => {
      console.log('TEST EMAIL:', options);
      return { messageId: 'test-' + Date.now() };
    }
  } as any;
} else {
  transporter = nodemailer.createTransport({
    host: process.env.SMTP_HOST,
    port: parseInt(process.env.SMTP_PORT || '587'),
    secure: process.env.SMTP_SECURE === 'true',
    auth: {
      user: process.env.SMTP_USER,
      pass: process.env.SMTP_PASSWORD
    }
  });
}

// Load and compile template
const loadTemplate = (templateName: string) => {
  const templatePath = path.join(__dirname, `${templateName}.hbs`);
  const template = fs.readFileSync(templatePath, 'utf-8');
  return Handlebars.compile(template);
};

// Send email
export async function sendEmail(
  to: string,
  subject: string,
  templateName: string,
  data: Record<string, any>
) {
  try {
    const template = loadTemplate(templateName);
    const html = template(data);

    const result = await transporter.sendMail({
      from: process.env.SMTP_FROM || 'noreply@example.com',
      to,
      subject,
      html
    });

    console.log(`Email sent to ${to}: ${result.messageId}`);
    return result;
  } catch (error) {
    console.error(`Failed to send email to ${to}:`, error);
    throw error;
  }
}

// Specific email senders
export async function sendOrderConfirmationEmail(userId: number, orderId: number) {
  const connection = await db.getConnection();
  
  try {
    const [users] = await connection.query('SELECT email FROM users WHERE id = ?', [userId]);
    const [orders] = await connection.query('SELECT * FROM orders WHERE id = ?', [orderId]);
    const [items] = await connection.query('SELECT * FROM order_items WHERE order_id = ?', [orderId]);
    
    connection.release();

    if (!users.length || !orders.length) return;

    const user = users[0];
    const order = orders[0];

    await sendEmail(
      user.email,
      `Order Confirmation #${order.id}`,
      'InvoiceCustomer',
      {
        customerName: user.first_name,
        orderId: order.id,
        orderNumber: order.order_number,
        items: items,
        totalAmount: order.total_amount,
        shippingAddress: order.shipping_address,
        orderUrl: `${process.env.BACKEND_PUBLIC_BASE_URL}/orders/${order.id}`
      }
    );
  } catch (error) {
    console.error('Failed to send order confirmation:', error);
  }
}

export async function sendShipmentNotification(orderId: number, trackingNumber: string) {
  const connection = await db.getConnection();
  
  try {
    const [orders] = await connection.query(
      `SELECT o.*, u.email, u.first_name FROM orders o 
       JOIN users u ON o.user_id = u.id WHERE o.id = ?`,
      [orderId]
    );
    
    connection.release();

    if (!orders.length) return;

    const order = orders[0];

    await sendEmail(
      order.email,
      `Your order #${order.id} has been shipped`,
      'Shipment',
      {
        customerName: order.first_name,
        orderId: order.id,
        trackingNumber,
        carrier: 'PostNord',
        trackingUrl: `https://tracking.postnord.com?ref=${trackingNumber}`
      }
    );
  } catch (error) {
    console.error('Failed to send shipment notification:', error);
  }
}
```

## Route Integration

```typescript
// Route that triggers email
app.post('/api/orders/:id/confirm', authenticateToken, async (req, res) => {
  try {
    const { id } = req.params;
    const userId = req.user?.id;

    // Update order status
    const connection = await db.getConnection();
    await connection.query(
      'UPDATE orders SET status = ? WHERE id = ? AND user_id = ?',
      ['confirmed', id, userId]
    );
    connection.release();

    // Send confirmation email
    await sendOrderConfirmationEmail(userId, parseInt(id));

    // Notify via WebSocket
    io.to(`order:${id}`).emit('order-updated', {
      orderId: id,
      status: 'confirmed'
    });

    res.json({ message: 'Order confirmed and email sent' });
  } catch (error) {
    console.error('Error confirming order:', error);
    res.status(500).json({ error: 'Failed to confirm order' });
  }
});
```

## Best Practices
✓ Always log payment transactions  
✓ Handle webhook signatures securely  
✓ Send confirmation emails asynchronously  
✓ Use templates for consistency  
✓ Test email with SMTP_TEST_MODE=true first  
✓ Implement payment retry logic  
✓ Store payment IDs for audit trail  
✓ Handle failed payments gracefully
